
### 1. 节点与局部连通

$$
\text{SameLevel}(u, v) \xLeftrightarrow{\text{def}} \text{Fun}(u) \land \text{Fun}(v) \land \text{Dom}(u) = \text{Dom}(v) \land \text{Ord}(\text{Dom}(u)) \land \text{Ran}(u) \subseteq 2 \land \text{Ran}(v) \subseteq 2
$$

$$
\text{ConnLevel}(u, v, E_{virt}) \xLeftrightarrow{\text{def}} \text{SameLevel}(u, v) \land \\
\forall C \in \mathcal{P}(2^{\text{Dom}(u)}) \Bigg( \Big( u \in C \land \forall x, y \in 2^{\text{Dom}(u)} \big( x \in C \land (\langle x,y \rangle \in E_{virt} \land \langle y,x \rangle \in E_{virt}) \to y \in C \big) \Big) \to v \in C \Bigg)
$$

### 2. 实步与虚步

$$
\text{RealStep}(u, v) \xLeftrightarrow{\text{def}} \text{Dom}(v) = \text{Dom}(u) \cup \{\text{Dom}(u)\} \land u \subset v
$$

$$
\text{VirtualStep}(u, v, E_{virt}) \xLeftrightarrow{\text{def}} \exists w \in 2^{\text{Dom}(u)} \Big( \langle u, w \rangle \in E_{virt} \land \text{RealStep}(w, v) \Big)
$$

### 3. 合法路径

$$
\text{ValidPath}(P, \lambda, E_{virt}) \xLeftrightarrow{\text{def}} \text{Fun}(P) \land \text{Dom}(P) = \lambda \land \text{Ord}(\lambda) \land \\
\forall \alpha \in \lambda \Big( \text{SameLevel}(P(\alpha), P(\alpha)) \land \text{Dom}(P(\alpha)) = \alpha \Big) \land \\
\forall \alpha \in \lambda \Big( \alpha \cup \{\alpha\} \in \lambda \to \big( \text{RealStep}(P(\alpha), P(\alpha \cup \{\alpha\})) \lor \text{VirtualStep}(P(\alpha), P(\alpha \cup \{\alpha\}), E_{virt}) \big) \Big) \land \\
\forall \alpha \in \lambda \Big( \alpha \cup \{\alpha\} \cup \{\alpha \cup \{\alpha\}\} \in \lambda \to \\
\neg \big( \text{VirtualStep}(P(\alpha), P(\alpha \cup \{\alpha\}), E_{virt}) \land \text{VirtualStep}(P(\alpha \cup \{\alpha\}), P(\alpha \cup \{\alpha\} \cup \{\alpha \cup \{\alpha\}\}), E_{virt}) \big) \Big) \land \\
\forall \gamma \in \lambda \Bigg( \text{Limit}(\gamma) \to \exists \beta \in \gamma \forall \alpha \in \gamma \setminus \beta \Big( \text{ConnLevel}(P(\gamma) \upharpoonright \alpha, P(\alpha), E_{virt}) \Big) \Bigg)
$$

### 4. 商空间

$$
\text{MacroEq}(p, q, \lambda, E_{virt}) \xLeftrightarrow{\text{def}} \exists \gamma \in \lambda \forall \delta \in \lambda \setminus \gamma \Big( \text{ConnLevel}(p(\delta), q(\delta), E_{virt}) \Big)
$$

$$
\text{QuotientSpace}(Q, \mathbf{P}, \lambda, E_{virt}) \xLeftrightarrow{\text{def}} Q = \Bigg\{ X \in \mathcal{P}(\mathbf{P}) \setminus \{\emptyset\} \mathrel{\Bigg|} \exists p \in \mathbf{P} \Big( X = \{ q \in \mathbf{P} \mid \text{MacroEq}(p, q, \lambda, E_{virt}) \} \Big) \Bigg\}
$$

### 5. 大边

$$
\text{MacroEdges}(\mathbf{E}_{\mathcal{G}}, Q, \lambda, E_{virt}) \xLeftrightarrow{\text{def}} \forall X \in Q \forall Y \in Q \Bigg( \langle X, Y \rangle \in \mathbf{E}_{\mathcal{G}} \leftrightarrow \exists p \in X \exists q \in Y \exists \gamma \in \lambda \forall \alpha \in \lambda \setminus \gamma \big( \langle p(\alpha), q(\alpha) \rangle \in E_{virt} \land \langle q(\alpha), p(\alpha) \rangle \notin E_{virt} \big) \Bigg)
$$

### 6. 图外延约束

$$
\text{GraphExt}(Q, \mathbf{E}_{\mathcal{G}}) \xLeftrightarrow{\text{def}} \forall a \in Q \forall b \in Q \Big( \forall c \in Q \big( \langle c, a \rangle \in \mathbf{E}_{\mathcal{G}} \leftrightarrow \langle c, b \rangle \in \mathbf{E}_{\mathcal{G}} \big) \to a = b \Big)
$$

$$
\text{Strict}(x, y, E_{virt}) \xLeftrightarrow{\text{def}} \langle x, y \rangle \in E_{virt} \land \langle y, x \rangle \notin E_{virt}
$$

$$
\text{IllFoundedLevel}(\alpha, E_{virt}) \xLeftrightarrow{\text{def}} \exists S \in \mathcal{P}(2^\alpha) \Bigg( S \neq \emptyset \land \forall y \in S \exists x \in S \exists u, v \in 2^\alpha \Big( \text{ConnLevel}(x, u, E_{virt}) \land \text{ConnLevel}(y, v, E_{virt}) \land \text{Strict}(u, v, E_{virt}) \Big) \Bigg)
$$

$$
\text{EventualLocalWF}(\lambda, E_{virt}) \xLeftrightarrow{\text{def}} \exists \beta \in \lambda \forall \alpha \in \lambda \setminus \beta \Big( \neg \text{IllFoundedLevel}(\alpha, E_{virt}) \Big)
$$

### 7. 树图宇宙表示问题

$$
\text{DenseWalk}(\mathbf{P}, E_{virt}) \xLeftrightarrow{\text{def}} \forall p \in \mathbf{P} \forall \alpha \in (\text{Dom}(p)\setminus \{0\} ) \exists q \in \mathbf{P} \Big( \langle p(\alpha), q(\alpha) \rangle \in E_{virt} \land \neg \text{ConnLevel}(p(\alpha), q(\alpha), \emptyset) \Big)
$$

$$
\text{DefinableRel}(E, M) \xLeftrightarrow{\text{def}} \exists \varphi \in \text{Fml}_{\mathcal{L}_\in} \exists p \in M \forall x, y \in M \Big( \langle x, y \rangle \in E \leftrightarrow \langle M, \in \rangle \models \varphi[x, y, p] \Big)
$$

$$
\mathbf{Question \ TGU} \xLeftrightarrow{\text{def}} \\ 
\exists \kappa \exists E_{virt} \exists \mathbf{P} \exists Q \exists \mathbf{E}_{\mathcal{G}} \exists f \Bigg( \text{Inaccessible}(\kappa) \land \text{DefinableRel}(E_{virt}, \mathcal{H}_\kappa) \land \text{DenseWalk}(\mathbf{P}, E_{virt}) \land \text{EventualLocalWF}(\kappa, E_{virt}) \land \\
\forall P \Big( P \in \mathbf{P} \leftrightarrow \text{ValidPath}(P, \kappa, E_{virt}) \Big) \land \\
\text{QuotientSpace}(Q, \mathbf{P}, \kappa, E_{virt}) \land \text{MacroEdges}(\mathbf{E}_{\mathcal{G}}, Q, \kappa, E_{virt}) \land \\
\text{GraphExt}(Q, \mathbf{E}_{\mathcal{G}}) \land \\
\text{Bijection}(f, Q, \mathcal{H}_\kappa) \land \forall x, y \in Q \big( \langle x, y \rangle \in \mathbf{E}_{\mathcal{G}} \leftrightarrow f(x) \in f(y) \big) \Bigg)
$$
