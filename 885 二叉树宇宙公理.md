
### 1. 节点与局部连通 (Nodes and Local Connectivity)

$$
\text{SameLevel}(u, v) \xLeftrightarrow{\text{def}} \exists \alpha \in \mathrm{Ord} \ \big( u \colon \alpha \to 2 \land v \colon \alpha \to 2 \big)
$$

$$
E_{virt}^s(x, y, E_{virt}) \xLeftrightarrow{\text{def}} \langle x, y \rangle \in E_{virt} \land \langle y, x \rangle \in E_{virt}
$$

$$
\text{ConnLevel}(u, v, E_{virt}) \xLeftrightarrow{\text{def}} \text{SameLevel}(u, v) \land \\
\forall C \subseteq {}^{\mathrm{dom}(u)}2 \ \Bigg( \Big( u \in C \land \forall x, y \in {}^{\mathrm{dom}(u)}2 \ \big( (x \in C \land E_{virt}^s(x, y, E_{virt})) \to y \in C \big) \Big) \to v \in C \Bigg)
$$

### 2. 实步与虚步 (Real Steps and Virtual Steps)

$$
\text{RealStep}(u, v) \xLeftrightarrow{\text{def}} \mathrm{dom}(v) = \mathrm{dom}(u) + 1 \land u \subset v
$$

$$
\text{VirtualStep}(u, v, E_{virt}) \xLeftrightarrow{\text{def}} \exists w \in {}^{\mathrm{dom}(u)}2 \ \Big( u \neq w \land \langle u, w \rangle \in E_{virt} \land \text{RealStep}(w, v) \Big)
$$

### 3. 合法路径 (Valid Paths)

$$
\text{ValidPath}(P, \lambda, E_{virt}) \xLeftrightarrow{\text{def}} \text{Fun}(P) \land \mathrm{dom}(P) = \lambda \land \mathrm{Ord}(\lambda) \land \\
\forall \alpha \in \lambda \Big( P(\alpha) \in {}^\alpha 2 \Big) \land \\
\forall \alpha \in \lambda \Big( \alpha+1 < \lambda \to \big( \text{RealStep}(P(\alpha), P(\alpha+1)) \lor \text{VirtualStep}(P(\alpha), P(\alpha+1), E_{virt}) \big) \Big) \land \\
\forall \alpha \in \lambda \Big( \alpha+2 < \lambda \to \neg \big( \text{VirtualStep}(P(\alpha), P(\alpha+1), E_{virt}) \land \text{VirtualStep}(P(\alpha+1), P(\alpha+2), E_{virt}) \big) \Big) \land \\
\forall \gamma \in \lambda \cap \mathrm{Lim} \ \Bigg( \exists \beta < \gamma \ \forall \alpha \in[\beta, \gamma) \Big( \text{ConnLevel}(P(\gamma) \restriction \alpha, P(\alpha), E_{virt}) \Big) \Bigg)
$$
*(注：$\mathrm{Lim}$ 表示极限序数类，$[\beta, \gamma)$ 表示序数区间。)*

### 4. 商空间 (Quotient Space)

$$
\text{MacroEq}(p, q, \lambda, E_{virt}) \xLeftrightarrow{\text{def}} \exists \gamma < \lambda \ \forall \delta \in[\gamma, \lambda) \Big( \text{ConnLevel}(p(\delta), q(\delta), E_{virt}) \Big)
$$

$$
\text{QuotientSpace}(Q, \mathbf{P}, \lambda, E_{virt}) \xLeftrightarrow{\text{def}} \\ 
Q = \Bigg\{ X \in \mathcal{P}(\mathbf{P}) \setminus \{\emptyset\} \mathrel{\Bigg|} \exists p \in \mathbf{P} \Big( X = \{ q \in \mathbf{P} \mid \text{MacroEq}(p, q, \lambda, E_{virt}) \} \Big) \Bigg\}
$$

### 5. 大边 (Macro Edges)与图外延

$$
\text{Strict}(x, y, E_{virt}) \xLeftrightarrow{\text{def}} \langle x, y \rangle \in E_{virt} \land \langle y, x \rangle \notin E_{virt}
$$

$$
\text{MacroEdges}(\mathbf{E}_{\mathcal{G}}, Q, \lambda, E_{virt}) \xLeftrightarrow{\text{def}} \mathbf{E}_{\mathcal{G}} \subseteq Q \times Q \land \\
\forall X, Y \in Q \Bigg( \langle X, Y \rangle \in \mathbf{E}_{\mathcal{G}} \leftrightarrow \exists p \in X \exists q \in Y \exists \gamma < \lambda \forall \alpha \in[\gamma, \lambda) \Big( \text{Strict}(p(\alpha), q(\alpha), E_{virt}) \Big) \Bigg)
$$

$$
\text{GraphExt}(Q, \mathbf{E}_{\mathcal{G}}) \xLeftrightarrow{\text{def}} \forall a, b \in Q \ \Bigg( \forall c \in Q \big( \langle c, a \rangle \in \mathbf{E}_{\mathcal{G}} \leftrightarrow \langle c, b \rangle \in \mathbf{E}_{\mathcal{G}} \big) \to a = b \Bigg)
$$

### 6. 良基性与稠密游走

$$
\text{IllFoundedLevel}(\alpha, E_{virt}) \xLeftrightarrow{\text{def}} \exists S \subseteq {}^\alpha 2 \Bigg( S \neq \emptyset \land \forall y \in S \ \exists x \in S \ \exists u, v \in {}^\alpha 2 \\
\Big( \text{ConnLevel}(x, u, E_{virt}) \land \text{ConnLevel}(y, v, E_{virt}) \land \text{Strict}(u, v, E_{virt}) \Big) \Bigg)
$$

$$
\text{EventualLocalWF}(\lambda, E_{virt}) \xLeftrightarrow{\text{def}} \exists \beta < \lambda \ \forall \alpha \in[\beta, \lambda) \Big( \neg \text{IllFoundedLevel}(\alpha, E_{virt}) \Big)
$$

$$
\text{DenseWalk}(\mathbf{P}, E_{virt}) \xLeftrightarrow{\text{def}} \forall p \in \mathbf{P} \forall \alpha \in \mathrm{dom}(p) \setminus \{0\} \ \exists q \in \mathbf{P} \Big( \langle p(\alpha), q(\alpha) \rangle \in E_{virt} \land p(\alpha) \neq q(\alpha) \Big)
$$

### 7. 无参数可定义性与TGU问题

$$
\text{ParamFreeDefinable}(E, M) \xLeftrightarrow{\text{def}} \exists \varphi(v_0, v_1) \in \text{Fml}_{\mathcal{L}_\in}^{\emptyset} \ \forall x, y \in M \Big( \langle x, y \rangle \in E \leftrightarrow \langle M, \in \rangle \models \varphi[x, y] \Big)
$$

$$
\mathbf{Question \ TGU} \xLeftrightarrow{\text{def}} \\
\exists \kappa \exists E_{virt} \exists \mathbf{P} \exists Q \exists \mathbf{E}_{\mathcal{G}} \exists f \Bigg( \text{Inaccessible}(\kappa) \land E_{virt} \subseteq \mathcal{H}_\kappa \times \mathcal{H}_\kappa \land \text{ParamFreeDefinable}(E_{virt}, \mathcal{H}_\kappa) \land \text{DenseWalk}(\mathbf{P}, E_{virt}) \land \text{EventualLocalWF}(\kappa, E_{virt}) \land \\
\mathbf{P} = \big\{ P \mid \text{ValidPath}(P, \kappa, E_{virt}) \big\} \land \\
\text{QuotientSpace}(Q, \mathbf{P}, \kappa, E_{virt}) \land \text{MacroEdges}(\mathbf{E}_{\mathcal{G}}, Q, \kappa, E_{virt}) \land \text{GraphExt}(Q, \mathbf{E}_{\mathcal{G}}) \land \\
\text{Bijection}(f, Q, \mathcal{H}_\kappa) \land \forall x, y \in Q \Big( \langle x, y \rangle \in \mathbf{E}_{\mathcal{G}} \leftrightarrow f(x) \in f(y) \Big) \Bigg)
$$
